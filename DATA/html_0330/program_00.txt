using UnityEngine;

public class Bullet : MonoBehaviour
{
    void OnCollisionEnter(Collision collision)
    {
        var hit = collision.gameObject;
        var enemy = hit.GetComponent&lt;Enemy&gt;();

        if (enemy == null)
            return;

        switch (hit.tag)
        {
            case "Shadow":
                enemy.ApplyDamageToShadow();
                break;
            case "Gigas":
                enemy.ApplyDamageToGigas();
                break;
        }
    }
}
