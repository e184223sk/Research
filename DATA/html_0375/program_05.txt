
// ------------値型------------
// データ型名　変数名
// int(整数)
int intNum = 2;

// float(単精度浮動小数点)
float floatNum = 1.234f;

// double(倍精度浮動小数点)
double doubleNum = 1.234;

// bool(論理)
// true か false 又は 1 か 0 です
bool boolVal = true;

// char(文字)
// ローマ字一文字を``で囲みます
char c = 'a';

// DateTime(日付)
DateTime date;


// ------------参照型------------

// String(文字列)
// 文字列を""で囲みます
String str = "hoge";

// 配列型
// 詳細は後述
int[] array;
