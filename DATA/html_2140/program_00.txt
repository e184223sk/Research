    public static ulong calcIndex(double lat, double lon)
    {
        uint _lat = (uint)((lat + 90.0) * (double)UInt32.MaxValue / 180.0);
        uint _lon = (uint)((lon + 180.0) * (double)UInt32.MaxValue / 360.0);

        ulong index = 0;

        byte x_byte0 = (byte)(_lon &amp; 0xFF);
        byte x_byte1 = (byte)((_lon &gt;&gt; 8) &amp; 0xFF);
        byte x_byte2 = (byte)((_lon &gt;&gt; 16) &amp; 0xFF);
        byte x_byte3 = (byte)((_lon &gt;&gt; 24) &amp; 0xFF);
        index |= (ulong)bit_table[x_byte3] &lt;&lt; 49;
        index |= (ulong)bit_table[x_byte2] &lt;&lt; 33;
        index |= (ulong)bit_table[x_byte1] &lt;&lt; 17;
        index |= (ulong)bit_table[x_byte0] &lt;&lt; 1;

        byte y_byte1 = (byte)((_lat &gt;&gt; 8) &amp; 0xFF);
        byte y_byte2 = (byte)((_lat &gt;&gt; 16) &amp; 0xFF);
        byte y_byte3 = (byte)((_lat &gt;&gt; 24) &amp; 0xFF);
        index |= (ulong)bit_table[y_byte3] &lt;&lt; 48;
        index |= (ulong)bit_table[y_byte2] &lt;&lt; 32;
        index |= (ulong)bit_table[y_byte1] &lt;&lt; 16;
        index |= (ulong)bit_table[y_byte0];
        return index;
    }
