    public static class Calc
    {
        /// &lt;summary&gt;足し算&lt;/summary&gt;
        public static int Add(int a, int b) { return a + b; }

        /// &lt;summary&gt;引き算&lt;/summary&gt;
        public static int Sub(int a, int b) { return a - b; }

        /// &lt;summary&gt;掛け算&lt;/summary&gt;
        public static int Multi(int a, int b) { return a * b; }

        /// &lt;summary&gt;割り算&lt;/summary&gt;
        public static int Div(int a, int b) { return a / b; }

        /// &lt;summary&gt;除算&lt;/summary&gt;
        public static int Mod(int a, int b) { return a % b; }
    }
