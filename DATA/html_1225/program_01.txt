
class NativeMethods {

    public const int DWMWA_EXTENDED_FRAME_BOUNDS = 9;

    [DllImport("user32.dll")]

        public int left; public int top; public int right; public int bottom;
    }

    public static extern IntPtr GetWindowDC(IntPtr hwnd);
