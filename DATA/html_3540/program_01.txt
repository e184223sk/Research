
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace LoopSample
{
    class Program
    {
        static void Main(string[] args)
        {
            int i = 0;
            while (i &lt; 5)
            {
                Console.WriteLine(i + ". Hello, world!");
                i++;
            }
        }
    }
}
