using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace LoopSample
{
    class Program
    {
        static void Main(string[] args)
        {
            for (int i = 0; i &lt; 5; i++)
            {
                Console.WriteLine(i + ". Hello, world!");
            }
        }
    }
}
