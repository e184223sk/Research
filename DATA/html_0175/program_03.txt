
    //配列の宣言
    int arrays [] = new int [] {1,2,3}; 

    //配列の長さ(大きさ)をコンソールに表示する。ここでは、「３」と表示される。
    Debug.Log(arrays.length); 

    //配列の番号を指定してコンソールに表示する。ここでは、「１」と表示される。
    Debug.Log(arrays[0]); 


    //配列の番号を指定して、内容を書き換える。
    arrays[0] = 900;

    //書き換えた内容の表示。ここでは、「900」と表示される。
    Debug.Log(arrays[0]);
