using UnityEngine;

public class Singleton&lt;T&gt; : MonoBehaviour where T : MonoBehaviour {
    static T instance;

    public static T Instance
    {
        get
        {
            if(instance == null)
            {
                instance = (T)FindObjectOfType(typeof(T));

                if(instance == null)
                {
                    Debug.LogError( typeof(T) + " none");
                }
            }

            return instance;
        }
    }
}
