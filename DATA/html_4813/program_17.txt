
using System;
using System.Linq;

namespace tdd_csharp.Lib {
    public class ColorConverter {

       // ç•¥

       public (int r, int g, int b) ToRgb(String hex) {
            string
                r = hex.Substring(1, 2),
                g = hex.Substring(3, 2),
                b = hex.Substring(5, 2);
            var rgb = new [] { r, g, b }
                .Select(s =&gt; Convert.ToInt32(s, 16))
                .ToList();
            return (rgb[0], rgb[1], rgb[2]);
        }
    }
}
