
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace OperatorSample
{
    class Program
    {
        static void Main(string[] args)
        {
            int a = 8;  // aは8．
            a += 5;  // aは13．
            a -= 5;  // aは8．
            a *= 5;  // aは40．
            a /= 5;  // aは8．
            a %= 5;  // aは3．
        }
    }
}
