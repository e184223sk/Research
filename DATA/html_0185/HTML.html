[C#] 非アクティブなTeamsアプリのミュート切り替え

C#Teams

0. 目的

非アクティブなTeamsアプリのミュート切り替えをショートカットで実行

1. 環境

  • OS : windows10 home(バージョン : 1909, OSビルド : 18363.1256)
  • 言語 : C#
  • コンパイラ : C:\Windows\Microsoft.NET\Framework64\v4.0.30319\csc.exe
  • Teams : 1.3.00.30866(64 ビット)

2. Teamsの設定

3. 流れ

  1. Teamsアプリで開かれた最新のウィンドウハンドルを取得(Activate_Teams.exe)
  2. ミュート切り替えのタイミングを指示(Activate.exe)
  3. ミュート切り替えの実施(Activate_Teams.exe)

4. 使い方

  • ProcessStart.exe : スタートアップアプリに登録し、PC起動時に実行(Activate_teams.exeがバックグラウンドで実行される)[1]
  • Activate.exe : ショートカットを作成後、プロパティから好きなショートカットキーを割り当てる(Activete.exeの実行により、ミュートが切り替わる)
    • "%APPDATA%\Microsoft\Windows\Start Menu\Programs\"(個別ユーザー用のスタートメニューフォルダ)にアクセスし、ショートカットを作成後、ショートカットキーを割り当てる[2]
    • ショートカットの実行が遅い場合はSysMainを停止[3]
    • Teamsウィンドウが最前面に来ないことがある場合は、「ForegroundLockTimeout」の設定を変更[4]

5. ソースコード

Activate_Teams.cs[5]
using System;
using System.IO;
using System.Linq;
using System.Text;
using System.Threading;
using System.Collections.Generic;
using System.Runtime.InteropServices;

class Search{

  public List<IntPtr> _windowHwnd { get; set; }
  public String _wihdowTitle { get; set; }
  public String _wihdowClass { get; set; }

  public delegate bool EnumWindowsDelegate(IntPtr hWnd, IntPtr lparam);

  [DllImport("user32.dll")]
  [return: MarshalAs(UnmanagedType.Bool)]
  public extern static bool EnumWindows(EnumWindowsDelegate lpEnumFunc, IntPtr lparam);

  [DllImport("user32.dll", CharSet = CharSet.Auto, SetLastError = true)]
  private static extern int GetWindowText(IntPtr hWnd, StringBuilder lpString, int nMaxCount);

  [DllImport("user32.dll", CharSet = CharSet.Auto, SetLastError = true)]
  private static extern int GetWindowTextLength(IntPtr hWnd);

  [DllImport("user32.dll", CharSet = CharSet.Auto, SetLastError = true)]
  private static extern int GetClassName(IntPtr hWnd, StringBuilder lpClassName, int nMaxCount);

  private bool EnumWindowCallBack(IntPtr hWnd, IntPtr lparam){

      // ウィンドウのタイトル長を取得
      int textLen = Search.GetWindowTextLength(hWnd);

      if (0 < textLen){
          // ウィンドウのタイトルを取得
          StringBuilder tsb = new StringBuilder(textLen + 1);
          Search.GetWindowText(hWnd, tsb, tsb.Capacity);

          // ウィンドウのクラス名を取得
          StringBuilder csb = new StringBuilder(256);
          Search.GetClassName(hWnd, csb, csb.Capacity);

          // 指定のタイトル名、クラス名のウィンドウハンドルを取得
          if(tsb.ToString().Contains(_wihdowTitle) && csb.ToString().Contains(_wihdowClass)){
              _windowHwnd.Add(hWnd);
          }
      }
      return true;
  }

  public void GetWindows(){
    _windowHwnd.Clear();
    EnumWindows(new EnumWindowsDelegate(EnumWindowCallBack),IntPtr.Zero);
  }

  public Search(){
    _windowHwnd = new List<IntPtr>();
  }

}

class Activate_Teams{

  [DllImport("user32.dll", CharSet = CharSet.Auto, SetLastError = true)]
  private static extern bool ShowWindowAsync(IntPtr hwnd, int nCmdShow);

  [DllImport("user32.dll", CharSet = CharSet.Auto, SetLastError = true)]
  private static extern bool SetForegroundWindow(IntPtr hWnd);

  public static void Main(string[] args)
  {
    var hWndList = new List<IntPtr>();
    String wihdowTitle = "| Microsoft Teams";
    String windowClass = "Chrome_WidgetWin_1";

    Search teams = new Search();
    teams._wihdowTitle = wihdowTitle;
    teams._wihdowClass = windowClass;

    while(true){

      teams.GetWindows();

      for(int i = 0; i < teams._windowHwnd.Count; i++ ){
        if(!hWndList.Contains(teams._windowHwnd[i])){
          hWndList.Add(teams._windowHwnd[i]); // 新しいウィドウハンドルをリストに追加
        }
      }

      for(int i = 0; i < hWndList.Count; i++){
        if(!teams._windowHwnd.Contains(hWndList[i])){
          hWndList.RemoveAt(i); // サーチできなかったウィドウハンドルをリストから削除
        }
      }

      if(File.Exists(args[0])){ // Activate.csによって作られたactivate.txtがあった場合
        ShowWindowAsync(hWndList.Last(), 4); // ウィンドウをアクティブ化
        SetForegroundWindow(hWndList.Last()); // ウィンドウを最前面へ
        System.Windows.Forms.SendKeys.SendWait("^+m"); // Ctrl+Shift+m(ミュート切り替え)

        File.Delete(args[0]); // activate.txtを削除
      }

      Thread.Sleep(100);
    }
  }
}
Activate.cs
using System;
using System.IO;
using System.Text;

class Activate{
    public static void Main(){
        if (!File.Exists(@"[Activate_Teams.exeを置いたフォルダのパス]\activate.txt")){
            File.Create(@"[Activate_Teams.exeを置いたフォルダのパス]\activate.txt").Close();
        }
    }
}
ProcessStart.cs
using System;
using System.Diagnostics;

class ProcessStart
{
    static void Main(string[] args)
    {
        var app = new ProcessStartInfo();
        app.FileName = @"[Activate_Teams.exeのパス]";
        app.Arguments = @"[Activate_Teams.exeを置いたフォルダのパス]\activate.txt";
        app.UseShellExecute = false;
        app.CreateNoWindow = true;

        Process.Start(app);
    }
}
